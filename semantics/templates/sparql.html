  Example Queries

  Example 1: Selecting first 25 triples

    SELECT ?subject ?predicate ?object 
    WHERE {
      ?subject ?predicate ?object
    }
    LIMIT 25
  
  Example 2: Selecting articles which includes Germany and Netherlands

    prefix dbo: <http://dbpedia.org/ontology/>
    prefix dcterms: <http://purl.org/dc/terms/>
    prefix foaf: <http://xmlns.com/foaf/0.1/>
    prefix sioc: <http://rdfs.org/sioc/ns#>
    
    SELECT ?country1 ?country2 ?article ?article_title WHERE { 
      ?article sioc:title ?article_title  .
      ?article sioc:topic ?article_section . 
      ?article sioc:content ?article_text . 
      ?article dbo:country ?country1 . FILTER (?country1 = "Germany") 
      ?article dbo:country ?country2 . FILTER (?country2 = "Netherlands") 
    } 
    LIMIT 100
